CREATE TABLE store_sku_check_stock_sku_data_result (
id int(11) NOT NULL AUTO_INCREMENT COMMENT ‘主键id’,
sku_data_id int(11) NOT NULL COMMENT ‘单品分析id’,
sku_result_type int(11) DEFAULT NULL COMMENT ‘单品分析原因’,
sku_result_value varchar(2000) DEFAULT NULL COMMENT ‘单品分析结果’,
valid int(11) DEFAULT NULL COMMENT ‘是否可用’,
create_time datetime NOT NULL COMMENT ‘创建时间’,
update_time datetime NOT NULL COMMENT ‘修改时间’,
PRIMARY KEY (id)
) ENGINE=InnoDB AUTO_INCREMENT=173 DEFAULT CHARSET=utf8mb4;

CREATE TABLE store_sku_check_stock_create_info (
id int(11) NOT NULL AUTO_INCREMENT COMMENT ‘主键id’,
create_id int(11) NOT NULL COMMENT ‘盘点分析id’,
check_info_key varchar(255) DEFAULT NULL COMMENT ‘分析类型’,
check_info_value varchar(2000) DEFAULT NULL COMMENT ‘分析原因’,
create_time datetime NOT NULL COMMENT ‘创建时间’,
update_time datetime NOT NULL COMMENT ‘修改时间’,
create_by varchar(45) DEFAULT NULL COMMENT ‘创建人’,
update_by varchar(45) DEFAULT NULL COMMENT ‘更新人’,
PRIMARY KEY (id)
) ENGINE=InnoDB AUTO_INCREMENT=33 DEFAULT CHARSET=utf8mb4;
